11               ((== x '>>>) (gui-make '(glue #t #f 15 0)))

13               (else

15        ((string? x) x)
16        ((and (pair? x) (ahash-ref gui-make-table (car x)))
17         (apply (car (ahash-ref gui-make-table (car x))) (list x)))
